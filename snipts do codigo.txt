# Verificar a versão do Windows Server
$osVersion = (Get-WmiObject -Class Win32_OperatingSystem).Caption

if ($osVersion -notmatch "Windows Server 2012" -and $osVersion -notmatch "Windows Server 2016") {
    Write-Host "Este script é compatível apenas com o Windows Server 2012 e 2016."
    exit
}





# Verificar se é o Windows Server 2012 ou 2016
if ($osVersion -notmatch '6\.2|6\.3') {
    Write-Host "Atenção: Este script foi otimizado para o Windows Server 2012 e 2016. Alguns recursos podem não funcionar corretamente nesta versão do sistema operacional."
    Write-Host
}

# Função para registrar informações de conexão bem-sucedida
Function Log-ConnectionInfo {
    param (
        [Parameter(Position = 0)]
        [string]$ComputerName,
        [Parameter(Position = 1)]
        [string]$Result
    )

    $timestamp = Get-Date -Format "yyyyMMddHHmmss"
    $logMessage = "$timestamp - $ComputerName - $Result"
    Write-Host $logMessage
}
